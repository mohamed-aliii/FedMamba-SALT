# ---------------------------------------------------------------------------
# Teacher backbone (MAE ViT-B/16)
# ---------------------------------------------------------------------------
# timm 0.3.2 is REQUIRED.  The MAE ViT-B/16 checkpoint was serialised with
# this version's VisionTransformer class definition.  Newer timm releases
# renamed / removed fields inside VisionTransformer (e.g., `patch_embed`,
# `cls_token`, `pos_embed`, head structure), so loading the checkpoint with
# timm >= 0.4 raises shape-mismatch or missing-key errors.
timm==0.3.2

# ---------------------------------------------------------------------------
# Student backbone (Inception-Mamba / selective-state-space layers)
# ---------------------------------------------------------------------------
# mamba-ssm — The Mamba CUDA kernel interface (selective_scan_fn,
# mamba_inner_fn).  On Google Colab, use --no-build-isolation so the
# build can see the existing PyTorch/CUDA.
# Tested with: mamba-ssm 1.2.0+ / causal-conv1d 1.4.0+
mamba-ssm>=1.2.0

# causal-conv1d — Hard build-time + runtime dependency of mamba-ssm.
# Must be installed BEFORE mamba-ssm.
# On Colab: pip install causal-conv1d>=1.4.0 --no-build-isolation
causal-conv1d>=1.4.0

# ---------------------------------------------------------------------------
# Tensor utilities
# ---------------------------------------------------------------------------
# einops — used for readable reshape / rearrange operations inside the
# Inception-Mamba blocks (e.g., patch un-embedding, channel mixing).
einops

# ---------------------------------------------------------------------------
# Scientific / training stack
# ---------------------------------------------------------------------------
numpy
Pillow
matplotlib
tqdm
PyYAML
scikit-image
pandas
